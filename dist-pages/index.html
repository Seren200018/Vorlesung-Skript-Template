    Herleitung Mehrmassenschwinger State Space

    Als Basis für die Zustandsraumdarstellung dient die Bewegungsgleichung des Mehrmassenschwingers.
    \begin{aligned} \overrightarrow{F}(t) &= \left[ M \right]\overrightarrow{\ddot{X}}+\left[ C \right]\overrightarrow{\dot{X}}+\left[ K \right]\overrightarrow{X}\\ \overrightarrow{\ddot{X}} &= -\left[ C \right]\left[ M \right]^{-1} \overrightarrow{\dot{X}}-\left[ K \right]\left[ M \right]^{-1}\overrightarrow{X}+\overrightarrow{F}(t)\left[ M \right]^{-1}\\ \end{aligned}
    Aus dieser Gleichung kann die Zustandsraum Darstellung hergeleitet werden.
    \begin{aligned} \overset{\dot{\vec{u}}(t+\Delta t)}{\begin{pmatrix} \ddot{\vec{x}}(t+\Delta t)\\ \dot{\vec{x}}(t+\Delta t) \end{pmatrix}}= \overset{\left[A\right]}{\begin{bmatrix} {-C\left[ M \right]^{-1}} & {-K\left[ M \right]^{-1}}\\ \left[ I \right]& \left[ 0 \right] \end{bmatrix} }\overset{\vec{u}(t)}{\begin{pmatrix} \dot{\vec{x}} \\ \vec{x} \end{pmatrix}} + \overset{\vec{F}_{SP}}{\begin{pmatrix}\vec{F}(t)\\\vec{0} \end{pmatrix}}\overset{\left[B\right]}{\begin{pmatrix} \left[M\right]^{-1}\\\left[0\right] \end{pmatrix}} \end{aligned}
    Mit der Rücktransformation der Zustandsvektoren: \begin{aligned} \begin{pmatrix} \dot{\vec{x}} \\ \vec{x} \end{pmatrix} &= \begin{bmatrix} I \end{bmatrix} \vec{u} \\ \end{aligned}

    import csv
from pathlib import Path
from statistics import mean


def load_temperatures(path: Path) -> list[float]:
    with path.open() as fh:
        reader = csv.DictReader(fh)
        return [float(row["temperatur"]) for row in reader if "temperatur" in row]


def summarize(values: list[float]) -> dict[str, float]:
    return {
        "anzahl": len(values),
        "minimum": min(values),
        "maximum": max(values),
        "mittelwert": mean(values),
    }


def main() -> None:
    data_file = Path("messwerte.csv")
    temps = load_temperatures(data_file)
    summary = summarize(temps)
    print("Temperaturauswertung")
    for key, val in summary.items():
        if isinstance(val, float):
            print(f"{key:>10}: {val:.2f}")
        else:
            print(f"{key:>10}: {val}")

    # Bewegter Durchschnitt über Fenster
    window = 5
    moving_avg = []
    for idx in range(len(temps)):
        start = max(0, idx - window + 1)
        chunk = temps[start : idx + 1]
        moving_avg.append(mean(chunk))
    print("\nGleitender Mittelwert (letzte 5 Messpunkte):")
    for i, val in enumerate(moving_avg, start=1):
        print(f"{i:>3}: {val:>7.2f}")

    # Einfache Anomalie-Erkennung
    threshold = summary["mittelwert"] + 2.5
    anomalies = [(i, t) for i, t in enumerate(temps, start=1) if t > threshold]
    if anomalies:
        print("\nAuffällige hohe Werte:")
        for idx, val in anomalies:
            print(f"  Messung {idx}: {val:.2f} °C")
    else:
        print("\nKeine auffälligen Werte gefunden.")

    # Export als einfache Textzusammenfassung
    report = Path("temperatur_report.txt")
    lines = [
        "Temperaturreport",
        f"Datei: {data_file.name}",
        f"Anzahl: {summary['anzahl']}",
        f"Minimum: {summary['minimum']:.2f}",
        f"Maximum: {summary['maximum']:.2f}",
        f"Mittelwert: {summary['mittelwert']:.2f}",
        "",
        "Gleitender Mittelwert:",
    ]
    lines.extend(f"{i}: {val:.2f}" for i, val in enumerate(moving_avg, start=1))
    lines.append("")
    lines.append("Anomalien:")
    if anomalies:
        lines.extend(f"{idx}: {val:.2f}" for idx, val in anomalies)
    else:
        lines.append("keine")
    report.write_text("\n".join(lines), encoding="utf-8")
    print(f"\nReport gespeichert in {report.resolve()}")


if __name__ == "__main__":
    main()
